/**
 * An individual user (i.e. conversation member).
 * @class
 * @param {Conversation} conversation
 * @param {Object} params 
 */
var Member = function(conversation, params, is_ourselves) {
    is_ourselves = is_ourselves || false;
    this.conversation = conversation;
    // Map the params
    for (var key in params) {
        this[key] = params[key];
    }
    this.updateMissingParams = function(params) {
        var self = this;
        if (params) {
            if (params.from) self.id = params.from;
            if (params.user && params.user.id) self.user_id = params.user.id;
            if (params.user && params.user.user_id) self.user_id = params.user.user_id;
            if (params.user && params.user.name) self.name = params.user.name;
            if (params.state) self.state = params.state;
            if (params.invited_by) self.invited_by = params.invited_by;
            if (params.timestamp) self.timestamp = params.timestamp;
            if (params.channel) self.channel = params.channel || {};
        }
    };
    // this._events = []; // A cache of events sent by this user
    // call WildEmitter constructor
    WildEmitter.call(this);

    //use this flag to specify member specific prototypes
    // if (is_ourselves) {

    // }
};

/*
 * Play the given stream only to this member within the conversation
 * 
 * @param {String} [params] 
 * 
 * @returns {Promise<Event>} - not tested yet
 */
Member.prototype.playStream = function(params) {
    var self = this;
    var p = new Promise(function(resolve, reject) {
        self.conversation.application.session.sendRequest({
            type: 'audio:play',
            cid: self.id,
            to: self.id,
            body: params
        }, function(response) {
            if (response.type === 'audio:play:success') {
                var event = new Event(self.conversation, response);
                resolve(event);
            } else {
                reject(new NexmoApiError(response));
            }
        });
    });
    return p;
};

/**
 * Kick this member from the conversation
 * 
 * @returns {Promise}
 */
Member.prototype.kick = function() {
    var self = this;
    var p = new Promise(function(resolve, reject) {
        self.conversation.application.session.sendRequest({
            type: 'conversation:member:delete',
            cid: self.conversation.id,
            from: self.conversation.me.id,
            to: self.id,
            body: {
                "member_id": self.id
            }
        }, function(response) {
            if (response.type === 'conversation:member:delete:success') {
                resolve(response.body);
            } else {
                reject(new NexmoApiError(response));
            }
        });
    });
    return p;
};

Member.prototype._handleEvent = function(event) {
    this.updateMissingParams(event.body);
    switch (event.type) {
        case 'member:invited':
            this.state = 'INVITED';
            this.emit(event.type, event.from, event);
            break;
        case 'member:joined':
            this.state = 'JOINED';
            this.emit(event.type, event.from, event);
            break;
        case 'member:left':
            this.state = 'LEFT';
            this.emit(event.type, event.from, event);
            break;
        case 'member:media':
            this.media = event.body;
            break;
        case 'text':
            var text_event = new TextEvent(this.conversation, event);
            this.emit(event.type, event.from, text_event); // This needs to generate a text event so we can send a "seen"
            break;
        case 'image':
            var image_event = new ImageEvent(this.conversation, event);
            this.emit(event.type, event.from, image_event); // This needs to generate an image event so we can send a "seen"
            break;
        default:
            this.emit(event.type, event.from, event.body);
            break;
    }
};
