var WildEmitter = require('wildemitter');

// Conversation Client Error
var ConversationClientError = require('./conversationClientError');

var NexmoApiError = ConversationClientError.NexmoApiError;
var NexmoClientError = ConversationClientError.NexmoClientError;

var User = function(application, params) {
    this.application = application;
    // Map the params
    for (var key in params) {
        this[key] = params[key];
    }
    // call WildEmitter constructor
    WildEmitter.call(this);
};


User.prototype = new WildEmitter();

/**
 * get Details of your user
 * @returns {Promise}
 */
User.prototype.getUser = function(user_id) {
    user_id = user_id || this.id;
    var self = this;
    var p = new Promise(function(resolve, reject) {
        var params = {
            user_id: user_id
        };
        self.application.session.sendRequest({
            type: 'user:get',
            from: self.application.me.id,
            body: params
        }, function(response) {
            if (response.type === 'user:get:success') {
                resolve(response.body);
            } else {
                reject(new NexmoApiError(response));
            }
        });
    });
    return p;
};

User.prototype._handleEvent = function(event) { };

module.exports = {
    User: User
};
