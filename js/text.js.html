var WildEmitter = require('wildemitter');
// Event = require('./event').Event;

// Conversation Client Error
var ConversationClientError = require('./conversationClientError');

var NexmoApiError = ConversationClientError.NexmoApiError;
var NexmoClientError = ConversationClientError.NexmoClientError;

/**
 * A text event
 *
 * @class Text
 */

var Text = function(conversation, params) {
    this.conversation = conversation;
    this.state = {
        seen_by: {},
        delivered_to: {}
    };
    this.timestamp = params.body.timestamp;
    if (params.body.replaces_id) {
        this.replaces_id = params.body.replaces_id;
        if (this.replaces_id in this.conversation.events) {
            this.conversation.events[this.replaces_id].replaced_by = params.id;
            this.conversation.events[this.replaces_id].text = "";
        }
    }
    // Map the params
    for (var key in params) {
        this[key] = params[key];
    }
    // call WildEmitter constructor
    WildEmitter.call(this);
};

Text.prototype = new WildEmitter();


/**
 * Set the message status to 'seen'
 * @returns {Promise}
 */
Text.prototype.seen = function() {
    var self = this;
    var p = new Promise(function(resolve, reject) {
        if (!self.state)
            self.state = {};
        if (self.conversation.me.id === self.from) {
            reject(new NexmoClientError("error:seen:own-message"));
        } else if (self.state.seen_by && self.state.seen_by[self.conversation.me.id]) {
            reject(new NexmoClientError("error:already-seen"));
        } else {
            var params = {
                event_id: self.id
            };
            self.conversation.application.session.sendRequest({
                type: 'text:seen',
                from: self.conversation.me.id,
                cid: self.conversation.id,
                body: params
            }, function(response) {
                if (response.type === 'text:seen:success') {
                    resolve();
                } else {
                    reject(new NexmoApiError(response));
                }
            });
        }
    });
    return p;
};


/**
 * Set the message status to 'delivered'. 
 * handled by the SDK
 * @returns {Promise}
 */
Text.prototype.delivered = function() {
    var self = this;
    var p = new Promise(function(resolve, reject) {
        if (self.conversation.me.id === self.from) {
            reject(new NexmoClientError("error:delivered:own-message"));
        } else if (self.state && self.state.delivered_to[self.conversation.me.id]) {
            reject(new NexmoClientError("error:already-delivered"));
        } else {
            var params = {
                event_id: self.id
            };
            self.conversation.application.session.sendRequest({
                type: 'text:delivered',
                from: self.conversation.me.id,
                cid: self.conversation.id,
                body: params
            }, function(response) {
                if (response.type === 'text:delivered:success') {
                    resolve();
                } else {
                    reject(new NexmoApiError(response));
                }
            });
            resolve();
        }
    });
    return p;
};

/**
 * Delete the event
 * @param {Object} [event_id=this.event_id]
 * @returns {Promise}
 * 
 * if the event_id param is not present, the "this" event will be default
 */
Text.prototype.del = function(event_id) {
    event_id = event_id || this.id;
    var self = this;
    var p = new Promise(function(resolve, reject) {
        self.conversation.application.session.sendRequest({
            type: 'event:delete',
            from: self.conversation.me.id,
            cid: self.conversation.id,
            body: {
                "event_id": event_id
            }
        }, function(response) {
            if (response.type === 'event:delete:success') {
                resolve();
            } else {
                reject(new NexmoApiError(response));
            }
        });
    });
    return p;
};

Text.prototype._handleEvent = function(event) {
    var self = this;
    switch (event.type) {
        default: self.emit(event.type, event.body);
        break;
    }
};
module.exports = {
    Text: Text
